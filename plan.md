import click
import pandas as pd
from ad_api.api.sp import Reports, Campaigns, AdGroups, Keywords
from ad_api.base import AdvertisingApiException


@click.group()
def cli():
    pass


@cli.command()
@click.option("--report_file", default="search_terms.csv")
def download_report(report_file):
    """
    Download search term report
    """

    report = Reports()

    report_data = report.post_report(
        body={
            "name": "searchTermReport",
            "startDate": "2026-01-01",
            "endDate": "2026-03-14",
            "configuration": {
                "adProduct": "SPONSORED_PRODUCTS",
                "reportTypeId": "spSearchTerm",
                "columns": [
                    "searchTerm",
                    "campaignName",
                    "impressions",
                    "clicks",
                    "sales",
                ],
                "timeUnit": "SUMMARY",
                "format": "GZIP_JSON"
            }
        }
    )

    print("Report requested:", report_data)


@cli.command()
@click.option("--campaign_name")
@click.option("--keywords_file", default="keywords.txt")
@click.option("--daily_budget", default=5)
def create_campaign(campaign_name, keywords_file, daily_budget):
    """
    Create manual campaign with keywords
    """

    campaigns = Campaigns()

    campaign = campaigns.create_campaigns(
        body=[{
            "name": campaign_name,
            "campaignType": "sponsoredProducts",
            "targetingType": "manual",
            "state": "enabled",
            "dailyBudget": daily_budget
        }]
    )

from sentence_transformers import SentenceTransformer
from sklearn.cluster import KMeans

model=SentenceTransformer("all-MiniLM-L6-v2")

def cluster_keywords(keywords,n=5):

    embeddings=model.encode(keywords)

    kmeans=KMeans(n_clusters=n)

    labels=kmeans.fit_predict(embeddings)

    clusters={}

    for k,l in zip(keywords,labels):

        clusters.setdefault(l,[]).append(k)

    return clusters
